def make_bricks(small, big, goal):
    small_len = small
    big_len = big*5
    sum_len = small_len + big_len
    if (small_len >= goal and small_len % goal == 0) or (big_len >= goal and big_len % goal ==0) or (sum_len >= goal and sum_len % goal == 0):
        return True
    else:
        return False
